cmake_minimum_required(VERSION 3.16)
project(CrossingGame VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# ── Tìm SFML ──────────────────────────────────────────────────────────────────
# Mac (Homebrew): brew install sfml@2
# Windows (vcpkg): vcpkg install sfml
find_package(SFML 2.5 COMPONENTS graphics window system audio REQUIRED)

# ── Thu thập toàn bộ .cpp trong src/ và subfolder ─────────────────────────────
file(GLOB_RECURSE SOURCES
    "src/*.cpp"
    "src/core/*.cpp"
    "src/obstacles/*.cpp"
    "src/animals/*.cpp"
    "src/levels/*.cpp"
    "src/ui/*.cpp"
)

# ── Tạo executable ────────────────────────────────────────────────────────────
add_executable(CrossingGame ${SOURCES})

# ── Include paths ─────────────────────────────────────────────────────────────
target_include_directories(CrossingGame PRIVATE
    src
    src/core
    src/obstacles
    src/animals
    src/levels
    src/ui
)

# ── Link SFML ─────────────────────────────────────────────────────────────────
target_link_libraries(CrossingGame PRIVATE
    sfml-graphics
    sfml-window
    sfml-system
    sfml-audio
)

# ── Copy assets vào thư mục build sau khi compile ────────────────────────────
add_custom_command(TARGET CrossingGame POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${CMAKE_SOURCE_DIR}/assets"
        "$<TARGET_FILE_DIR:CrossingGame>/assets"
    COMMENT "Copying assets to build directory..."
)

# ── Copy saves vào thư mục build ─────────────────────────────────────────────
add_custom_command(TARGET CrossingGame POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory
        "$<TARGET_FILE_DIR:CrossingGame>/saves"
    COMMENT "Creating saves directory..."
)

# ── Tắt warning SFML trên macOS Xcode mới ────────────────────────────────────
if(APPLE)
    target_compile_options(CrossingGame PRIVATE -w)
endif()

# ── Windows: copy SFML DLL vào cùng thư mục exe ──────────────────────────────
if(WIN32)
    add_custom_command(TARGET CrossingGame POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            $<TARGET_RUNTIME_DLLS:CrossingGame>
            $<TARGET_FILE_DIR:CrossingGame>
        COMMAND_EXPAND_LISTS
    )
endif()
